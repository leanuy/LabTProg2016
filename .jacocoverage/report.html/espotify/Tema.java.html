    private int num;
    private Album album;

        return nombre;
    }
    public int getDuracion() {
        return duracion;
    }
    public int getNum() {
        return num;
    }
    public String getNombreAlbum()
    {
        return album.getNombre();
    }

    DataTema getData() {
        return new DataTema(nombre, duracion, num);
    }
}
